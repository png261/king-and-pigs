77void Box2D::createCollisionObject(const int width, const int height, const b2Vec2 position)
78{
79 b2BodyDef body;

82 b2Body* const groundBody = this->getWorld()->CreateBody(&body);

91 vertices[0].Set(-w / 2, -h / 2);
92 vertices[1].Set(w / 2, -h / 2);
93 vertices[2].Set(w / 2, h / 2);
94 vertices[3].Set(-w / 2, h / 2);
95 box.CreateLoop(vertices, 4);
