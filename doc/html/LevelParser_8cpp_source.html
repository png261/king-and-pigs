18Level* LevelParser::parseLevel(const char* levelFile)
19{
20 XMLDocument levelDocument;
21 if (levelDocument.LoadFile(levelFile) != XML_SUCCESS) {
22 Log::error(std::string("fail to load level file: ") + levelFile);
23 }
24
25 Level* const pLevel = new Level();
26 XMLElement* const pRoot = levelDocument.RootElement();
27 if (pRoot == nullptr) {
28 Log::error(std::string(levelFile) + ": missing root element");
29 }
30
31 m_tileSize = std::stoi(pRoot->Attribute("tilewidth"));
32 m_width = std::stoi(pRoot->Attribute("width"));
33 m_height = std::stoi(pRoot->Attribute("height"));
34
35 for (XMLElement* e = pRoot->FirstChildElement(); e != nullptr; e = e->NextSiblingElement()) {
36 if (e->Value() == std::string("properties")) {
37 for (XMLElement* pProperty = e->FirstChildElement(); pProperty != nullptr;
38 pProperty = pProperty->NextSiblingElement()) {
39 if (pProperty->Value() == std::string("property")) {
40 parseTextures(pProperty);
41 }
42 }
43 }
44 }
45
46 for (XMLElement* e = pRoot->FirstChildElement(); e != nullptr; e = e->NextSiblingElement()) {
47 if (e->Value() == std::string("tileset")) {
48 parseTilesets(e, pLevel->getTilesets(), pLevel->getCollisionShapes());
49 }
50 }
51
52 for (XMLElement* e = pRoot->FirstChildElement(); e != nullptr; e = e->NextSiblingElement()) {
53 if (e->Value() == std::string("objectgroup") || e->Value() == std::string("layer")) {
54 bool isTileLayer =
55 e->FirstChildElement()->Value() == std::string("data") ||
56 (e->FirstChildElement()->NextSiblingElement() != 0 &&
57 e->FirstChildElement()->NextSiblingElement()->Value() == std::string("data"));
58 bool isObjectLayer = e->FirstChildElement()->Value() == std::string("object");
59
60 if (isObjectLayer) {
61 parseObjectLayer(e, pLevel->getLayers(), pLevel);
62 } else if (isTileLayer) {
63 parseTileLayer(
64 e,
65 pLevel->getLayers(),
66 pLevel->getTilesets(),
67 pLevel->getCollisionShapes());
68 }
69 }
70 }
71
72 return pLevel;
73}
