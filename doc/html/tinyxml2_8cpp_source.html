206char* StrPair::ParseText(char* p, const char* endTag, int strFlags, int* curLineNumPtr)
207{
208 TIXMLASSERT(p);
209 TIXMLASSERT(endTag && *endTag);
210 TIXMLASSERT(curLineNumPtr);
211
212 char* start = p;
213 const char endChar = *endTag;
214 size_t length = strlen(endTag);
215
216 // Inner loop of text parsing.
217 while (*p) {
218 if (*p == endChar && strncmp(p, endTag, length) == 0) {
219 Set(start, p, strFlags);
220 return p + length;
221 } else if (*p == '\n') {
222 ++(*curLineNumPtr);
223 }
224 ++p;
225 TIXMLASSERT(p);
226 }
227 return 0;

249void StrPair::CollapseWhitespace()
250{
251 // Adjusting _start would cause undefined behavior on delete[]
252 TIXMLASSERT((_flags & NEEDS_DELETE) == 0);
253 // Trim leading space.
254 _start = XMLUtil::SkipWhiteSpace(_start, 0);
255
256 if (*_start) {
257 const char* p = _start; // the read pointer
258 char* q = _start; // the write pointer
259
260 while (*p) {
261 if (XMLUtil::IsWhiteSpace(*p)) {
262 p = XMLUtil::SkipWhiteSpace(p, 0);
263 if (*p == 0) {
264 break; // don't write to q; this trims the trailing space.
265 }
266 *q = ' ';
267 ++q;
268 }
269 *q = *p;
270 ++q;
271 ++p;
272 }
273 *q = 0;
274 }

277const char* StrPair::GetStr()
278{
279 TIXMLASSERT(_start);
280 TIXMLASSERT(_end);
281 if (_flags & NEEDS_FLUSH) {
282 *_end = 0;
283 _flags ^= NEEDS_FLUSH;
284
285 if (_flags) {
286 const char* p = _start; // the read pointer
287 char* q = _start; // the write pointer
288
289 while (p < _end) {
290 if ((_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == CR) {
291 // CR-LF pair becomes LF
292 // CR alone becomes LF
293 // LF-CR becomes LF
294 if (*(p + 1) == LF) {
295 p += 2;
296 } else {
297 ++p;
298 }
299 *q = LF;
300 ++q;
301 } else if ((_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == LF) {
302 if (*(p + 1) == CR) {
303 p += 2;
304 } else {
305 ++p;
306 }
307 *q = LF;
308 ++q;
309 } else if ((_flags & NEEDS_ENTITY_PROCESSING) && *p == '&') {
310 // Entities handled by tinyXML2:
311 // - special entities in the entity table [in/out]
312 // - numeric character reference [in]
313 // &#20013; or &#x4e2d;
314
315 if (*(p + 1) == '#') {
316 const int buflen = 10;
317 char buf[buflen] = {0};
318 int len = 0;
319 const char* adjusted =
320 const_cast<char*>(XMLUtil::GetCharacterRef(p, buf, &len));
321 if (adjusted == 0) {
322 *q = *p;
323 ++p;
324 ++q;
325 } else {
326 TIXMLASSERT(0 <= len && len <= buflen);
327 TIXMLASSERT(q + len <= adjusted);
328 p = adjusted;
329 memcpy(q, buf, len);
330 q += len;
331 }
332 } else {
333 bool entityFound = false;
334 for (int i = 0; i < NUM_ENTITIES; ++i) {
335 const Entity& entity = entities[i];
336 if (strncmp(p + 1, entity.pattern, entity.length) == 0 &&
337 *(p + entity.length + 1) == ';') {
338 // Found an entity - convert.
339 *q = entity.value;
340 ++q;
341 p += entity.length + 2;
342 entityFound = true;
343 break;
344 }
345 }
346 if (!entityFound) {
347 // fixme: treat as error?
348 ++p;
349 ++q;
350 }
351 }
352 } else {
353 *q = *p;
354 ++p;
355 ++q;
356 }
357 }
358 *q = 0;
359 }
360 // The loop below has plenty going on, and this
361 // is a less useful mode. Break it out.
362 if (_flags & NEEDS_WHITESPACE_COLLAPSING) {
363 CollapseWhitespace();
364 }
365 _flags = (_flags & NEEDS_DELETE);
366 }
367 TIXMLASSERT(_start);
368 return _start;
369}

401void XMLUtil::ConvertUTF32ToUTF8(unsigned long input, char* output, int* length)
402{
403 const unsigned long BYTE_MASK = 0xBF;
404 const unsigned long BYTE_MARK = 0x80;
405 const unsigned long FIRST_BYTE_MARK[7] = {0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC};
406
407 if (input < 0x80) {
408 *length = 1;
409 } else if (input < 0x800) {
410 *length = 2;
411 } else if (input < 0x10000) {
412 *length = 3;
413 } else if (input < 0x200000) {
414 *length = 4;
415 } else {
416 *length = 0; // This code won't convert this correctly anyway.
417 return;
418 }
419
420 output += *length;
421
422 // Scary scary fall throughs are annotated with carefully designed comments
423 // to suppress compiler warnings such as -Wimplicit-fallthrough in gcc
424 switch (*length) {
425 case 4:
426 --output;
427 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
428 input >>= 6;
429 // fall through
430 case 3:
431 --output;
432 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
433 input >>= 6;
434 // fall through
435 case 2:
436 --output;
437 *output = static_cast<char>((input | BYTE_MARK) & BYTE_MASK);
438 input >>= 6;
439 // fall through
440 case 1:
441 --output;
442 *output = static_cast<char>(input | FIRST_BYTE_MARK[*length]);
443 break;
444 default: TIXMLASSERT(false);
445 }
446}

448const char* XMLUtil::GetCharacterRef(const char* p, char* value, int* length)
449{
450 // Presume an entity, and pull it out.
451 *length = 0;
452
453 if (*(p + 1) == '#' && *(p + 2)) {
454 unsigned long ucs = 0;
455 TIXMLASSERT(sizeof(ucs) >= 4);
456 ptrdiff_t delta = 0;
457 unsigned mult = 1;
458 static const char SEMICOLON = ';';
459
460 if (*(p + 2) == 'x') {
461 // Hexadecimal.
462 const char* q = p + 3;
463 if (!(*q)) {
464 return 0;
465 }
466
467 q = strchr(q, SEMICOLON);
468
469 if (!q) {
470 return 0;
471 }
472 TIXMLASSERT(*q == SEMICOLON);
473
474 delta = q - p;
475 --q;
476
477 while (*q != 'x') {
478 unsigned int digit = 0;
479
480 if (*q >= '0' && *q <= '9') {
481 digit = *q - '0';
482 } else if (*q >= 'a' && *q <= 'f') {
483 digit = *q - 'a' + 10;
484 } else if (*q >= 'A' && *q <= 'F') {
485 digit = *q - 'A' + 10;
486 } else {
487 return 0;
488 }
489 TIXMLASSERT(digit < 16);
490 TIXMLASSERT(digit == 0 || mult <= UINT_MAX / digit);
491 const unsigned int digitScaled = mult * digit;
492 TIXMLASSERT(ucs <= ULONG_MAX - digitScaled);
493 ucs += digitScaled;
494 TIXMLASSERT(mult <= UINT_MAX / 16);
495 mult *= 16;
496 --q;
497 }
498 } else {
499 // Decimal.
500 const char* q = p + 2;
501 if (!(*q)) {
502 return 0;
503 }
504
505 q = strchr(q, SEMICOLON);
506
507 if (!q) {
508 return 0;
509 }
510 TIXMLASSERT(*q == SEMICOLON);
511
512 delta = q - p;
513 --q;
514
515 while (*q != '#') {
516 if (*q >= '0' && *q <= '9') {
517 const unsigned int digit = *q - '0';
518 TIXMLASSERT(digit < 10);
519 TIXMLASSERT(digit == 0 || mult <= UINT_MAX / digit);
520 const unsigned int digitScaled = mult * digit;
521 TIXMLASSERT(ucs <= ULONG_MAX - digitScaled);
522 ucs += digitScaled;
523 } else {
524 return 0;
525 }
526 TIXMLASSERT(mult <= UINT_MAX / 10);
527 mult *= 10;
528 --q;
529 }
530 }
531 // convert the UCS to UTF-8
532 ConvertUTF32ToUTF8(ucs, value, length);
533 return p + delta + 1;
534 }
535 return p + 1;
536}

672char* XMLDocument::Identify(char* p, XMLNode** node)
673{
674 TIXMLASSERT(node);
675 TIXMLASSERT(p);
676 char* const start = p;
677 int const startLine = _parseCurLineNum;
678 p = XMLUtil::SkipWhiteSpace(p, &_parseCurLineNum);
679 if (!*p) {
680 *node = 0;
681 TIXMLASSERT(p);
682 return p;
683 }
684
685 // These strings define the matching patterns:
686 static const char* xmlHeader = {"<?"};
687 static const char* commentHeader = {"<!--"};
688 static const char* cdataHeader = {"<![CDATA["};
689 static const char* dtdHeader = {"<!"};
690 static const char* elementHeader = {"<"}; // and a header for everything else; check last.
691
692 static const int xmlHeaderLen = 2;
693 static const int commentHeaderLen = 4;
694 static const int cdataHeaderLen = 9;
695 static const int dtdHeaderLen = 2;
696 static const int elementHeaderLen = 1;
697
698 TIXMLASSERT(sizeof(XMLComment) == sizeof(XMLUnknown)); // use same memory pool
699 TIXMLASSERT(sizeof(XMLComment) == sizeof(XMLDeclaration)); // use same memory pool
700 XMLNode* returnNode = 0;
701 if (XMLUtil::StringEqual(p, xmlHeader, xmlHeaderLen)) {
702 returnNode = CreateUnlinkedNode<XMLDeclaration>(_commentPool);
703 returnNode->_parseLineNum = _parseCurLineNum;
704 p += xmlHeaderLen;
705 } else if (XMLUtil::StringEqual(p, commentHeader, commentHeaderLen)) {
706 returnNode = CreateUnlinkedNode<XMLComment>(_commentPool);
707 returnNode->_parseLineNum = _parseCurLineNum;
708 p += commentHeaderLen;
709 } else if (XMLUtil::StringEqual(p, cdataHeader, cdataHeaderLen)) {
710 XMLText* text = CreateUnlinkedNode<XMLText>(_textPool);
711 returnNode = text;
712 returnNode->_parseLineNum = _parseCurLineNum;
713 p += cdataHeaderLen;
714 text->SetCData(true);
715 } else if (XMLUtil::StringEqual(p, dtdHeader, dtdHeaderLen)) {
716 returnNode = CreateUnlinkedNode<XMLUnknown>(_commentPool);
717 returnNode->_parseLineNum = _parseCurLineNum;
718 p += dtdHeaderLen;
719 } else if (XMLUtil::StringEqual(p, elementHeader, elementHeaderLen)) {
720 returnNode = CreateUnlinkedNode<XMLElement>(_elementPool);
721 returnNode->_parseLineNum = _parseCurLineNum;
722 p += elementHeaderLen;
723 } else {
724 returnNode = CreateUnlinkedNode<XMLText>(_textPool);
725 returnNode->_parseLineNum =
726 _parseCurLineNum; // Report line of first non-whitespace character
727 p = start; // Back it up, all the text counts.
728 _parseCurLineNum = startLine;
729 }
730
731 TIXMLASSERT(returnNode);
732 TIXMLASSERT(p);
733 *node = returnNode;
734 return p;
735}

982char* XMLNode::ParseDeep(char* p, StrPair* parentEndTag, int* curLineNumPtr)
983{
984 // This is a recursive method, but thinking about it "at the current level"
985 // it is a pretty simple flat list:
986 // <foo/>
987 // <!-- comment -->
988 //
989 // With a special case:
990 // <foo>
991 // </foo>
992 // <!-- comment -->
993 //
994 // Where the closing element (/foo) *must* be the next thing after the
995 // opening element, and the names must match. BUT the tricky bit is that the
996 // closing element will be read by the child.
997 //
998 // 'endTag' is the end tag for this node, it is returned by a call to a
999 // child. 'parentEnd' is the end tag for the parent, which is filled in and
1000 // returned.
1001
1002 XMLDocument::DepthTracker tracker(_document);
1003 if (_document->Error()) return 0;
1004
1005 while (p && *p) {
1006 XMLNode* node = 0;
1007
1008 p = _document->Identify(p, &node);
1009 TIXMLASSERT(p);
1010 if (node == 0) {
1011 break;
1012 }
1013
1014 const int initialLineNum = node->_parseLineNum;
1015
1016 StrPair endTag;
1017 p = node->ParseDeep(p, &endTag, curLineNumPtr);
1018 if (!p) {
1019 _document->DeleteNode(node);
1020 if (!_document->Error()) {
1021 _document->SetError(XML_ERROR_PARSING, initialLineNum, 0);
1022 }
1023 break;
1024 }
1025
1026 const XMLDeclaration* const decl = node->ToDeclaration();
1027 if (decl) {
1028 // Declarations are only allowed at document level
1029 //
1030 // Multiple declarations are allowed but all declarations
1031 // must occur before anything else.
1032 //
1033 // Optimized due to a security test case. If the first node is
1034 // a declaration, and the last node is a declaration, then only
1035 // declarations have so far been added.
1036 bool wellLocated = false;
1037
1038 if (ToDocument()) {
1039 if (FirstChild()) {
1040 wellLocated = FirstChild() && FirstChild()->ToDeclaration() && LastChild() &&
1041 LastChild()->ToDeclaration();
1042 } else {
1043 wellLocated = true;
1044 }
1045 }
1046 if (!wellLocated) {
1047 _document->SetError(
1048 XML_ERROR_PARSING_DECLARATION,
1049 initialLineNum,
1050 "XMLDeclaration value=%s",
1051 decl->Value());
1052 _document->DeleteNode(node);
1053 break;
1054 }
1055 }
1056
1057 XMLElement* ele = node->ToElement();
1058 if (ele) {
1059 // We read the end tag. Return it to the parent.
1060 if (ele->ClosingType() == XMLElement::CLOSING) {
1061 if (parentEndTag) {
1062 ele->_value.TransferTo(parentEndTag);
1063 }
1064 node->_memPool->SetTracked(); // created and then immediately deleted.
1065 DeleteNode(node);
1066 return p;
1067 }
1068
1069 // Handle an end tag returned to this level.
1070 // And handle a bunch of annoying errors.
1071 bool mismatch = false;
1072 if (endTag.Empty()) {
1073 if (ele->ClosingType() == XMLElement::OPEN) {
1074 mismatch = true;
1075 }
1076 } else {
1077 if (ele->ClosingType() != XMLElement::OPEN) {
1078 mismatch = true;
1079 } else if (!XMLUtil::StringEqual(endTag.GetStr(), ele->Name())) {
1080 mismatch = true;
1081 }
1082 }
1083 if (mismatch) {
1084 _document->SetError(
1085 XML_ERROR_MISMATCHED_ELEMENT,
1086 initialLineNum,
1087 "XMLElement name=%s",
1088 ele->Name());
1089 _document->DeleteNode(node);
1090 break;
1091 }
1092 }
1093 InsertEndChild(node);
1094 }
1095 return 0;
1096}

1812char* XMLElement::ParseAttributes(char* p, int* curLineNumPtr)
1813{
1814 XMLAttribute* prevAttribute = 0;
1815
1816 // Read the attributes.
1817 while (p) {
1818 p = XMLUtil::SkipWhiteSpace(p, curLineNumPtr);
1819 if (!(*p)) {
1820 _document
1821 ->SetError(XML_ERROR_PARSING_ELEMENT, _parseLineNum, "XMLElement name=%s", Name());
1822 return 0;
1823 }
1824
1825 // attribute.
1826 if (XMLUtil::IsNameStartChar((unsigned char)*p)) {
1827 XMLAttribute* attrib = CreateAttribute();
1828 TIXMLASSERT(attrib);
1829 attrib->_parseLineNum = _document->_parseCurLineNum;
1830
1831 const int attrLineNum = attrib->_parseLineNum;
1832
1833 p = attrib->ParseDeep(p, _document->ProcessEntities(), curLineNumPtr);
1834 if (!p || Attribute(attrib->Name())) {
1835 DeleteAttribute(attrib);
1836 _document->SetError(
1837 XML_ERROR_PARSING_ATTRIBUTE,
1838 attrLineNum,
1839 "XMLElement name=%s",
1840 Name());
1841 return 0;
1842 }
1843 // There is a minor bug here: if the attribute in the source xml
1844 // document is duplicated, it will not be detected and the
1845 // attribute will be doubly added. However, tracking the
1846 // 'prevAttribute' avoids re-scanning the attribute list. Preferring
1847 // performance for now, may reconsider in the future.
1848 if (prevAttribute) {
1849 TIXMLASSERT(prevAttribute->_next == 0);
1850 prevAttribute->_next = attrib;
1851 } else {
1852 TIXMLASSERT(_rootAttribute == 0);
1853 _rootAttribute = attrib;
1854 }
1855 prevAttribute = attrib;
1856 }
1857 // end of the tag
1858 else if (*p == '>') {
1859 ++p;
1860 break;
1861 }
1862 // end of the tag
1863 else if (*p == '/' && *(p + 1) == '>') {
1864 _closingType = CLOSED;
1865 return p + 2; // done; sealed element.
1866 } else {
1867 _document->SetError(XML_ERROR_PARSING_ELEMENT, _parseLineNum, 0);
1868 return 0;
1869 }
1870 }
1871 return p;
1872}

1928char* XMLElement::ParseDeep(char* p, StrPair* parentEndTag, int* curLineNumPtr)
1929{

1934 // parsed just like a regular element then deleted from

1948 return p;
1949 }

2497void XMLPrinter::PrintString(const char* p, bool restricted)
2498{
2499 // Look for runs of bytes between entities to print.
2500 const char* q = p;
2501
2502 if (_processEntities) {
2503 const bool* flag = restricted ? _restrictedEntityFlag : _entityFlag;
2504 while (*q) {
2505 TIXMLASSERT(p <= q);
2506 // Remember, char is sometimes signed. (How many times has that
2507 // bitten me?)
2508 if (*q > 0 && *q < ENTITY_RANGE) {
2509 // Check for entities. If one is found, flush
2510 // the stream up until the entity, write the
2511 // entity, and keep looking.
2512 if (flag[static_cast<unsigned char>(*q)]) {
2513 while (p < q) {
2514 const size_t delta = q - p;
2515 const int toPrint = (INT_MAX < delta) ? INT_MAX : static_cast<int>(delta);
2516 Write(p, toPrint);
2517 p += toPrint;
2518 }
2519 bool entityPatternPrinted = false;
2520 for (int i = 0; i < NUM_ENTITIES; ++i) {
2521 if (entities[i].value == *q) {
2522 Putc('&');
2523 Write(entities[i].pattern, entities[i].length);
2524 Putc(';');
2525 entityPatternPrinted = true;
2526 break;
2527 }
2528 }
2529 if (!entityPatternPrinted) {
2530 // TIXMLASSERT( entityPatternPrinted ) causes gcc
2531 // -Wunused-but-set-variable in release
2532 TIXMLASSERT(false);
2533 }
2534 ++p;
2535 }
2536 }
2537 ++q;
2538 TIXMLASSERT(p <= q);
2539 }
2540 // Flush the remaining string. This will be the entire
2541 // string if an entity wasn't found.
2542 if (p < q) {
2543 const size_t delta = q - p;
2544 const int toPrint = (INT_MAX < delta) ? INT_MAX : static_cast<int>(delta);
2545 Write(p, toPrint);
2546 }
2547 } else {
2548 Write(p);
2549 }
2550}
