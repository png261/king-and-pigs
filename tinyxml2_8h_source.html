209/*
210 A dynamic array of Plain Old Data. Doesn't support constructors, etc.
211 Has a small initial memory pool, so that low or no usage will not
212 cause a call to new/delete
213*/
214template <class T, int INITIAL_SIZE>
215class DynArray
216{
217public:
218 DynArray()
219 : _mem(_pool)
220 , _allocated(INITIAL_SIZE)
221 , _size(0)
222 {}
223
224 ~DynArray()
225 {
226 if (_mem != _pool) {
227 delete[] _mem;
228 }
229 }
230
231 void Clear() { _size = 0; }
232
233 void Push(T t)
234 {
235 TIXMLASSERT(_size < INT_MAX);
236 EnsureCapacity(_size + 1);
237 _mem[_size] = t;
238 ++_size;
239 }
240
241 T* PushArr(int count)
242 {
243 TIXMLASSERT(count >= 0);
244 TIXMLASSERT(_size <= INT_MAX - count);
245 EnsureCapacity(_size + count);
246 T* ret = &_mem[_size];
247 _size += count;
248 return ret;
249 }
250
251 T Pop()
252 {
253 TIXMLASSERT(_size > 0);
254 --_size;
255 return _mem[_size];
256 }
257
258 void PopArr(int count)
259 {
260 TIXMLASSERT(_size >= count);
261 _size -= count;
262 }
263
264 bool Empty() const { return _size == 0; }
265
266 T& operator[](int i)
267 {
268 TIXMLASSERT(i >= 0 && i < _size);
269 return _mem[i];
270 }
271
272 const T& operator[](int i) const
273 {
274 TIXMLASSERT(i >= 0 && i < _size);
275 return _mem[i];
276 }
277
278 const T& PeekTop() const
279 {
280 TIXMLASSERT(_size > 0);
281 return _mem[_size - 1];
282 }
283
284 int Size() const
285 {
286 TIXMLASSERT(_size >= 0);
287 return _size;
288 }
289
290 int Capacity() const
291 {
292 TIXMLASSERT(_allocated >= INITIAL_SIZE);
293 return _allocated;
294 }
295
296 void SwapRemove(int i)
297 {
298 TIXMLASSERT(i >= 0 && i < _size);
299 TIXMLASSERT(_size > 0);
300 _mem[i] = _mem[_size - 1];
301 --_size;
302 }
303
304 const T* Mem() const
305 {
306 TIXMLASSERT(_mem);
307 return _mem;
308 }
309
310 T* Mem()
311 {
312 TIXMLASSERT(_mem);
313 return _mem;
314 }
315
316private:
317 DynArray(const DynArray&); // not supported
318 void operator=(const DynArray&); // not supported
319
320 void EnsureCapacity(int cap)
321 {
322 TIXMLASSERT(cap > 0);
323 if (cap > _allocated) {
324 TIXMLASSERT(cap <= INT_MAX / 2);
325 const int newAllocated = cap * 2;
326 T* newMem = new T[newAllocated];
327 TIXMLASSERT(newAllocated >= _size);
328 memcpy(
329 newMem,
330 _mem,
331 sizeof(T) * _size); // warning: not using constructors, only
332 // works for PODs
333 if (_mem != _pool) {
334 delete[] _mem;
335 }
336 _mem = newMem;
337 _allocated = newAllocated;
338 }
339 }
340
341 T* _mem;
342 T _pool[INITIAL_SIZE];
343 int _allocated; // objects allocated
344 int _size; // number objects in use
345};

363/*
364 Template child class to create pools of the correct type.
365*/
366template <int ITEM_SIZE>
367class MemPoolT : public MemPool
368{
369public:
370 MemPoolT()
371 : _blockPtrs()
372 , _root(0)
373 , _currentAllocs(0)
374 , _nAllocs(0)
375 , _maxAllocs(0)
376 , _nUntracked(0)
377 {}
378 ~MemPoolT() { MemPoolT<ITEM_SIZE>::Clear(); }
379
380 void Clear()
381 {
382 // Delete the blocks.
383 while (!_blockPtrs.Empty()) {
384 Block* lastBlock = _blockPtrs.Pop();
385 delete lastBlock;
386 }
387 _root = 0;
388 _currentAllocs = 0;
389 _nAllocs = 0;
390 _maxAllocs = 0;
391 _nUntracked = 0;
392 }
393
394 virtual int ItemSize() const { return ITEM_SIZE; }
395 int CurrentAllocs() const { return _currentAllocs; }
396
397 virtual void* Alloc()
398 {
399 if (!_root) {
400 // Need a new block.
401 Block* block = new Block;
402 _blockPtrs.Push(block);
403
404 Item* blockItems = block->items;
405 for (int i = 0; i < ITEMS_PER_BLOCK - 1; ++i) {
406 blockItems[i].next = &(blockItems[i + 1]);
407 }
408 blockItems[ITEMS_PER_BLOCK - 1].next = 0;
409 _root = blockItems;
410 }
411 Item* const result = _root;
412 TIXMLASSERT(result != 0);
413 _root = _root->next;
414
415 ++_currentAllocs;
416 if (_currentAllocs > _maxAllocs) {
417 _maxAllocs = _currentAllocs;
418 }
419 ++_nAllocs;
420 ++_nUntracked;
421 return result;
422 }
423
424 virtual void Free(void* mem)
425 {
426 if (!mem) {
427 return;
428 }
429 --_currentAllocs;
430 Item* item = static_cast<Item*>(mem);
431#ifdef TINYXML2_DEBUG
432 memset(item, 0xfe, sizeof(*item));
433#endif
434 item->next = _root;
435 _root = item;
436 }
437 void Trace(const char* name)
438 {
439 printf(
440 "Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d "
441 "blocks=%d\n",
442 name,
443 _maxAllocs,
444 _maxAllocs * ITEM_SIZE / 1024,
445 _currentAllocs,
446 ITEM_SIZE,
447 _nAllocs,
448 _blockPtrs.Size());
449 }
450
451 void SetTracked()
452 {
453 --_nUntracked;
454 }
455
456 int Untracked() const
457 {
458 return _nUntracked;
459 }

461 // This number is perf sensitive. 4k seems like a good tradeoff on my
462 // machine. The test file is large, 170k. Release: VS2010 gcc(no
463 // opt)
464 // 1k: 4000
465 // 2k: 4000
466 // 4k: 3900 21000
467 // 16k: 5200
468 // 32k: 4300
469 // 64k: 4000 21000
470 // Declared public because some compilers do not accept to use
471 // ITEMS_PER_BLOCK in private part if ITEMS_PER_BLOCK is private
472 enum { ITEMS_PER_BLOCK = (4 * 1024) / ITEM_SIZE };
473
474private:
475 MemPoolT(const MemPoolT&); // not supported
476 void operator=(const MemPoolT&); // not supported
477
478 union Item {
479 Item* next;
480 char itemData[ITEM_SIZE];
481 };
482 struct Block
483 {
484 Item items[ITEMS_PER_BLOCK];
485 };
486 DynArray<Block*, 10> _blockPtrs;
487 Item* _root;
488
489 int _currentAllocs;
490 int _nAllocs;
491 int _maxAllocs;
492 int _nUntracked;
493};
