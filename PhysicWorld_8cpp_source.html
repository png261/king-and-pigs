154void PhysicWorld::DoorInListener(b2Contact* contact)
155{
156 b2Fixture* fixtureA = contact->GetFixtureA();
157 b2Fixture* fixtureB = contact->GetFixtureB();
158 uint16 catA = fixtureA->GetFilterData().categoryBits;
159 uint16 catB = fixtureB->GetFilterData().categoryBits;
160
161 bool isPlayerDoor = catA == CAT_DOOR_IN && catB == CAT_PLAYER;
162 if (!isPlayerDoor) {
163 return;
164 }
165
166 DoorIn* door = dynamic_cast<DoorIn*>((GameObject*)(fixtureA->GetBody()->GetUserData().pointer));
167 Player* player =
168 dynamic_cast<Player*>((GameObject*)(fixtureB->GetBody()->GetUserData().pointer));
169
170 if (player == nullptr || door == nullptr) {
171 return;
172 }
173
174 if (player->isWantDoorIn()) {
175 if (door->isOpened()) {
176 Game::Instance()->nextLevel();
177 } else {
178 door->open();
179 player->doorIn();
180 }
181 }

184void PhysicWorld::AttackListener(b2Contact* contact)
185{
186 b2Fixture* const A = contact->GetFixtureA();
187 b2Fixture* const B = contact->GetFixtureB();
188 uint16 const catA = A->GetFilterData().categoryBits;
189 uint16 const catB = B->GetFilterData().categoryBits;
190
191 bool isAttack = ((catA | catB) == (CAT_ATTACK_SENSOR | CAT_ENEMY)) ||
192 ((catA | catB) == (CAT_ATTACK_SENSOR | CAT_PLAYER));
193
194 if (!isAttack) {
195 return;
196 }
197
198 if (catA == CAT_ATTACK_SENSOR) {
199 handleAttack(A, B);
200 } else {
201 handleAttack(B, A);
202 }
203}
